52  /**
53   * Contains invocation parameters for SmartSprites, provides methods for validating the parameters.
54   */
55  public final class SmartSpritesParameters {
56      /**
57       * Path to the directory that contains the css files to be processed. Directories containing CSS and image files
58       * must be writable, unless output.dir.path is provided. The root.dir.path can be either absolute, e.g.
59       * c:/myproject/web or relative to the directory in which this script is run.
60       */
61      @Option(name = "--root-dir-path", required = false, metaVar = "DIR")
62      private String rootDir;
63  
64      /**
65       * Paths to individual CSS files to process. Either {@link #rootDir} or {@link #cssFiles} must be not empty. If only
66       * {@link #cssFiles} is not empty, {@link #outputDir} must be blank. If both {@link #rootDir} and {@link #cssFiles}
67       * are not empty, {@link #outputDir} is supported but only {@link #cssFiles} from {@link #rootDir} are processed.
68       */
69      @Argument(metaVar = "CSS-FILES")
70      @Option(name = "--css-files", required = false, metaVar = "FILES")
71      private List<String> cssFiles;
72  
73      /**
74       * Output directory for processed CSS files and CSS-relative sprite images. The directory structure relative to
75       * root.dir.path will be preserved in the output directory. E.g. if CSS files are contained in the css/base
76       * directory of root.dir.path, the processed results will be written to output.dir.path/css/base. Also, CSS-relative
77       * sprite images will be written to the output directory. Sprite images with document-root-relative URLs will be
78       * written relative to the document.root.dir.path.
79       * <p>
80       * If the output.dir.path directory does not exist, it will be created.
81       * <p>
82       * You can leave this property empty, in which case the CSS files will be written next to the original CSS files
83       * with css.file.suffix, and sprite images will be written relative to CSS files.
84       * <p>
85       * If you are using a non-empty output.dir.path, you might want to use an empty css.file.suffix.
86       */
87      @Option(name = "--output-dir-path", metaVar = "DIR")
88      private String outputDir;
89  
90      /**
91       * Document root path for document-root-relative (starting with '/') image urls in CSS. All such image URLs will be
92       * taken relative to document.root.dir.path. Also document-root-relative sprite URLs will be written relative to
93       * document.root.dir.path. You can leave this property empty if your CSS uses only CSS-relative image URLs. *
94       */
95      @Option(name = "--document-root-dir-path", metaVar = "DIR")
96      private String documentRootDir;
97

99       * Message logging level. If you're getting lots of INFO messages and want to see only warnings, set this option to

103     private MessageLevel logLevel;

142     /** By default, we don't generate sprite directive in output css. */
143     public static final boolean DEFAULT_MARK_SPRITE_IMAGES = false;

239     /**
240      * Validates the provided parameters. All resource paths are resolved against the local file system.
241      *
242      * @param log
243      *            the log
244      *
245      * @return <code>true</code> if the parameters are valid
246      */
247     public boolean validate(MessageLog log) {
248         boolean valid = true;
249 
250         // Either root dir or css files are required
251         if (!hasRootDir() && !hasCssFiles()) {
252             log.error(MessageType.EITHER_ROOT_DIR_OR_CSS_FILES_IS_REQUIRED);
253             return false;
254         }
255 
256         // If there is no output dir, we can't have both root dir or css files
257         if (!hasOutputDir() && hasRootDir() && hasCssFiles()) {
258             log.error(MessageType.ROOT_DIR_AND_CSS_FILES_CANNOT_BE_BOTH_SPECIFIED_UNLESS_WITH_OUTPUT_DIR);
259             return false;
260         }
261 
262         // Check root dir if provided
263         if (hasRootDir()) {
264             final File rootDir = FileUtils.getCanonicalOrAbsoluteFile(this.rootDir);
265             if (!rootDir.exists() || !rootDir.isDirectory()) {
266                 log.error(MessageType.ROOT_DIR_DOES_NOT_EXIST_OR_IS_NOT_DIRECTORY, this.rootDir);
267                 valid = false;
268             }
269         }
270 
271         // Check output dir if provided
272         if (hasOutputDir()) {
273             // For output dir, we need root dir
274             if (!hasRootDir()) {
275                 log.error(MessageType.ROOT_DIR_IS_REQUIRED_FOR_OUTPUT_DIR);
276                 return false;
277             }
278 
279             final File outputDir = FileUtils.getCanonicalOrAbsoluteFile(this.outputDir);
280             if (outputDir.exists() && !outputDir.isDirectory()) {
281                 log.error(MessageType.OUTPUT_DIR_IS_NOT_DIRECTORY, this.outputDir);
282                 valid = false;
283             }
284         }
285 
286         if (!hasOutputDir() && StringUtils.isBlank(cssFileSuffix)) {
287             log.error(MessageType.CSS_FILE_SUFFIX_IS_REQUIRED_IF_NO_OUTPUT_DIR);
288             valid = false;
289         }
290 
291         if (hasDocumentRootDir()) {
292             final File documentRootDir = FileUtils.getCanonicalOrAbsoluteFile(this.documentRootDir);
293             if (!documentRootDir.exists() || !documentRootDir.isDirectory()) {
294                 log.error(MessageType.DOCUMENT_ROOT_DIR_DOES_NOT_EXIST_OR_IS_NOT_DIRECTORY, this.documentRootDir);
295                 valid = false;
296             }
297         }
298 
299         return valid;
300     }

302     /**
303      * Gets the css file suffix.
304      *
305      * @param suffix
306      *            the suffix
307      *
308      * @return the css file suffix
309      */
310     private String getCssFileSuffix(String suffix) {
311         if (suffix != null) {
312             return suffix;
313         }
314         if (!hasOutputDir()) {
315             // If there is no output dir, we must have some suffix
316             return DEFAULT_CSS_FILE_SUFFIX;
317         }
318         // If we have an output dir, we can have an empty suffix
319         return "";
320     }
