37  package org.carrot2.labs.smartsprites;
38  
39  import com.google.common.hash.Hashing;
40  
41  import java.awt.image.BufferedImage;
42  import java.io.IOException;
43  import java.util.Map;
44  import java.util.regex.Matcher;
45  import java.util.regex.Pattern;
46  
47  import org.carrot2.labs.smartsprites.SpriteImageDirective.SpriteUidType;
48  
49  /**
50   * A merged sprite image consisting of a number of individual images.
51   */
52  public class SpriteImage {
53  
54      /** The rendered sprite image bitmap. */
55      public final BufferedImage sprite;
56  
57      /**
58       * All {@link SpriteReferenceReplacement}s corresponding to the individual images this sprite image consists of.
59       */
60      public final Map<SpriteReferenceOccurrence, SpriteReferenceReplacement> spriteReferenceReplacements;
61  
62      /**
63       * {@link SpriteImageOccurrence} for which this {@link SpriteImage} has been built.
64       */
65      public final SpriteImageOccurrence spriteImageOccurrence;
66  
67      /**
68       * The {@link SpriteImageDirective#imagePath} with variables resolved.
69       */
70      public String resolvedPath;
71  
72      /**
73       * The width of the final sprite.
74       */
75      public int spriteWidth;
76  
77      /**
78       * The height of the final sprite.
79       */
80      public int spriteHeight;
81  
82      /**
83       * The scale to apply to the final sprite's background-size.
84       */
85      public float scaleRatio;
86  
87      /** The Constant SPRITE_VARIABLE. */
88      private static final Pattern SPRITE_VARIABLE = Pattern.compile("${sprite}", Pattern.LITERAL);
89  
90      /**
91       * Instantiates a new sprite image.
92       *
93       * @param sprite
94       *            the sprite
95       * @param spriteImageOccurrence
96       *            the sprite image occurrence
97       * @param spriteReplacements
98       *            the sprite replacements
99       * @param width
100      *            the width
101      * @param height
102      *            the height
103      * @param scale
104      *            the scale
105      */
106     public SpriteImage(BufferedImage sprite, SpriteImageOccurrence spriteImageOccurrence,
107             Map<SpriteReferenceOccurrence, SpriteReferenceReplacement> spriteReplacements, int width, int height,
108             float scale) {
109         this.sprite = sprite;
110         this.spriteReferenceReplacements = spriteReplacements;
111         this.spriteImageOccurrence = spriteImageOccurrence;
112         this.spriteWidth = width;
113         this.spriteHeight = height;
114         this.scaleRatio = scale;
115 
116         for (SpriteReferenceReplacement replacement : spriteReplacements.values()) {
117             replacement.spriteImage = this;
118         }
119     }
120 
121     /**
122      * Resolve image path.
123      *
124      * @param image
125      *            the image
126      * @param timestamp
127      *            the timestamp
128      *
129      * @return the string
130      *
131      * @throws IOException
132      *             Signals that an I/O exception has occurred.
133      */
134     String resolveImagePath(byte[] image, String timestamp) throws IOException {
135         String imagePath = spriteImageOccurrence.spriteImageDirective.imagePath;
136 
137         // Backwards compatibility: if there are no place holders in the path
138         // and the UID type is defined, append the UID as a query string just like
139         // the previous versions did. To be removed in 0.4.0.
140         if (spriteImageOccurrence.spriteImageDirective.uidType != SpriteUidType.NONE
141                 && !SpriteUidType.SHA512.pattern.matcher(imagePath).find()
142                 && !SpriteUidType.DATE.pattern.matcher(imagePath).find()) {
143             imagePath += "?${" + spriteImageOccurrence.spriteImageDirective.uidType.toString() + "}";
144         }
145 
146         // Resolve SHA512 hash
147         Matcher sha512Matcher = SpriteUidType.SHA512.pattern.matcher(imagePath);
148         if (sha512Matcher.find()) {
149             // Compute SHA512 only when necessary
150             imagePath = sha512Matcher.replaceAll(computeSha512(image));
151         }
152 
153         // Resolve timestamp
154         imagePath = SpriteUidType.DATE.pattern.matcher(imagePath).replaceAll(timestamp);
155 
156         // Resolve sprite name
157         imagePath = SPRITE_VARIABLE.matcher(imagePath).replaceAll(spriteImageOccurrence.spriteImageDirective.spriteId);
158 
159         this.resolvedPath = imagePath;
160         return this.resolvedPath;
161     }
162 
163     /**
164      * Computes Sha512 using guava.
165      *
166      * @param image
167      *            the image
168      *
169      * @return the string
170      */
171     private static String computeSha512(byte[] image) {
172         return Hashing.sha512().hashBytes(image).toString();
173     }
174 }
