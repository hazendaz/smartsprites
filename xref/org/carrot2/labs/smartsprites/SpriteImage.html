8   package org.carrot2.labs.smartsprites;
9   
10  import com.google.common.hash.Hashing;
11  
12  import java.awt.image.BufferedImage;
13  import java.io.IOException;
14  import java.util.Map;
15  import java.util.regex.Matcher;
16  import java.util.regex.Pattern;
17  
18  import org.carrot2.labs.smartsprites.SpriteImageDirective.SpriteUidType;
19  
20  /**
21   * A merged sprite image consisting of a number of individual images.
22   */
23  public class SpriteImage {
24  
25      /** The rendered sprite image bitmap. */
26      public final BufferedImage sprite;
27  
28      /**
29       * All {@link SpriteReferenceReplacement}s corresponding to the individual images this sprite image consists of.
30       */
31      public final Map<SpriteReferenceOccurrence, SpriteReferenceReplacement> spriteReferenceReplacements;
32  
33      /**
34       * {@link SpriteImageOccurrence} for which this {@link SpriteImage} has been built.
35       */
36      public final SpriteImageOccurrence spriteImageOccurrence;
37  
38      /**
39       * The {@link SpriteImageDirective#imagePath} with variables resolved.
40       */
41      public String resolvedPath;
42  
43      /**
44       * The width of the final sprite.
45       */
46      public int spriteWidth;
47  
48      /**
49       * The height of the final sprite.
50       */
51      public int spriteHeight;
52  
53      /**
54       * The scale to apply to the final sprite's background-size.
55       */
56      public float scaleRatio;
57  
58      /** The Constant SPRITE_VARIABLE. */
59      private static final Pattern SPRITE_VARIABLE = Pattern.compile("${sprite}", Pattern.LITERAL);
60  
61      /**
62       * Instantiates a new sprite image.
63       *
64       * @param sprite
65       *            the sprite
66       * @param spriteImageOccurrence
67       *            the sprite image occurrence
68       * @param spriteReplacements
69       *            the sprite replacements
70       * @param width
71       *            the width
72       * @param height
73       *            the height
74       * @param scale
75       *            the scale
76       */
77      public SpriteImage(BufferedImage sprite, SpriteImageOccurrence spriteImageOccurrence,
78              Map<SpriteReferenceOccurrence, SpriteReferenceReplacement> spriteReplacements, int width, int height,
79              float scale) {
80          this.sprite = sprite;
81          this.spriteReferenceReplacements = spriteReplacements;
82          this.spriteImageOccurrence = spriteImageOccurrence;
83          this.spriteWidth = width;
84          this.spriteHeight = height;
85          this.scaleRatio = scale;
86  
87          for (SpriteReferenceReplacement replacement : spriteReplacements.values()) {
88              replacement.spriteImage = this;
89          }
90      }
91  
92      /**
93       * Resolve image path.
94       *
95       * @param image
96       *            the image
97       * @param timestamp
98       *            the timestamp
99       *
100      * @return the string
101      *
102      * @throws IOException
103      *             Signals that an I/O exception has occurred.
104      */
105     String resolveImagePath(byte[] image, String timestamp) throws IOException {
106         String imagePath = spriteImageOccurrence.spriteImageDirective.imagePath;
107 
108         // Backwards compatibility: if there are no place holders in the path
109         // and the UID type is defined, append the UID as a query string just like
110         // the previous versions did. To be removed in 0.4.0.
111         if (spriteImageOccurrence.spriteImageDirective.uidType != SpriteUidType.NONE
112                 && !SpriteUidType.SHA512.pattern.matcher(imagePath).find()
113                 && !SpriteUidType.DATE.pattern.matcher(imagePath).find()) {
114             imagePath += "?${" + spriteImageOccurrence.spriteImageDirective.uidType.toString() + "}";
115         }
116 
117         // Resolve SHA512 hash
118         Matcher sha512Matcher = SpriteUidType.SHA512.pattern.matcher(imagePath);
119         if (sha512Matcher.find()) {
120             // Compute SHA512 only when necessary
121             imagePath = sha512Matcher.replaceAll(computeSha512(image));
122         }
123 
124         // Resolve timestamp
125         imagePath = SpriteUidType.DATE.pattern.matcher(imagePath).replaceAll(timestamp);
126 
127         // Resolve sprite name
128         imagePath = SPRITE_VARIABLE.matcher(imagePath).replaceAll(spriteImageOccurrence.spriteImageDirective.spriteId);
129 
130         this.resolvedPath = imagePath;
131         return this.resolvedPath;
132     }
133 
134     /**
135      * Computes Sha512 using guava.
136      *
137      * @param image
138      *            the image
139      *
140      * @return the string
141      */
142     private static String computeSha512(byte[] image) {
143         return Hashing.sha512().hashBytes(image).toString();
144     }
145 }
