8   package org.carrot2.labs.smartsprites.message;
9   
10  import java.io.PrintStream;
11  
12  import org.carrot2.labs.smartsprites.message.Message.MessageLevel;
13  
14  /**
15   * Outputs logged {@link Message}s to the provided {@link PrintStream}. The {@link Message#toString()} method will be
16   * used to format the messages.
17   */
18  public class PrintStreamMessageSink implements MessageSink {
19      /** The stream to print messages to. */
20      private final PrintStream printStream;
21  
22      /** Logging level. */
23      private final MessageLevel level;
24  
25      /**
26       * Creates a {@link PrintStreamMessageSink} with the provided <code>printStream</code> and {@link MessageLevel#INFO}
27       * logging level.
28       *
29       * @param printStream
30       *            the print stream
31       */
32      public PrintStreamMessageSink(PrintStream printStream) {
33          this(printStream, MessageLevel.INFO);
34      }
35  
36      /**
37       * Creates a {@link PrintStreamMessageSink} with the provided <code>printStream</code> and logging
38       * <code>level</code>.
39       *
40       * @param printStream
41       *            the print stream
42       * @param level
43       *            the level
44       */
45      public PrintStreamMessageSink(PrintStream printStream, MessageLevel level) {
46          this.printStream = printStream;
47          this.level = level;
48      }
49  
50      @Override
51      public void add(Message message) {
52          if (MessageLevel.COMPARATOR.compare(message.level, level) >= 0) {
53              printStream.println(message.toString());
54          }
55      }
56  }
