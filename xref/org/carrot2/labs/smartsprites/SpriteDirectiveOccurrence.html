8   package org.carrot2.labs.smartsprites;
9   
10  /**
11   * A base class for SmartSprites directives.
12   */
13  public class SpriteDirectiveOccurrence {
14  
15      /** CSS file in which this directive was found. */
16      public final String cssFile;
17  
18      /** Line number on which the directive occurred. */
19      public final int line;
20  
21      /**
22       * Instantiates a new sprite directive occurrence.
23       *
24       * @param cssFile
25       *            the css file
26       * @param line
27       *            the line
28       */
29      public SpriteDirectiveOccurrence(String cssFile, int line) {
30          this.cssFile = cssFile;
31          this.line = line;
32      }
33  }
