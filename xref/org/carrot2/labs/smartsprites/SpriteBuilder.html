75      /**
76       * Creates a {@link SpriteBuilder} with the provided parameters and log.
77       *
78       * @param parameters
79       *            the parameters
80       * @param messageLog
81       *            the message log
82       * @param resourceHandler
83       *            the resource handler
84       */
85      public SpriteBuilder(SmartSpritesParameters parameters, MessageLog messageLog, ResourceHandler resourceHandler) {
86          this.messageLog = messageLog;
87          this.parameters = parameters;
88          this.resourceHandler = resourceHandler;
89          spriteDirectiveOccurrenceCollector = new SpriteDirectiveOccurrenceCollector(messageLog, resourceHandler);
90          spriteImageBuilder = new SpriteImageBuilder(parameters, messageLog, resourceHandler);
91      }

93      /**
94       * Performs processing for this builder's parameters. This method resolves all paths against the local file system.
95       *
96       * @throws IOException
97       *             Signals that an I/O exception has occurred.
98       */
99      public void buildSprites() throws IOException {
100         if (!parameters.validate(messageLog)) {
101             return;
102         }
103 
104         final Collection<String> filePaths;
105         if (parameters.getCssFiles() != null && !parameters.getCssFiles().isEmpty()) {
106             // Take directly provided css fle paths
107             filePaths = parameters.getCssFiles();
108 
109             // If root dir is provided, filter out those files that are outside root dir
110             if (StringUtils.isNotBlank(parameters.getOutputDir())) {
111                 filterFilesOutsideRootDir(filePaths);
112             }
113 
114             // Make sure the files exist and are really files
115             for (Iterator<String> it = filePaths.iterator(); it.hasNext();) {
116                 final String path = it.next();
117                 final File file = Path.of(path).toFile();
118                 if (file.exists()) {
119                     if (!file.isFile()) {
120                         messageLog.warning(MessageType.CSS_PATH_IS_NOT_A_FILE, path);
121                         it.remove();
122                     }
123                 } else {
124                     messageLog.warning(MessageType.CSS_FILE_DOES_NOT_EXIST, path);
125                     it.remove();
126                 }
127             }
128         } else {
129             // Take all css files from the root dir
130             final List<File> files = Lists.newArrayList(org.apache.commons.io.FileUtils
131                     .listFiles(parameters.getRootDirFile(), new String[] { "css" }, true));
132             files.sort(Comparator.comparing(File::getAbsolutePath));
133 
134             filePaths = new ArrayList<>();
135             for (File file : files) {
136                 filePaths.add(file.getPath());
137             }
138         }
139 
140         buildSprites(filePaths);
141     }

143     /**
144      * Filter files outside root dir.
145      *
146      * @param filePaths
147      *            the file paths
148      *
149      * @throws IOException
150      *             Signals that an I/O exception has occurred.
151      */
152     private void filterFilesOutsideRootDir(Collection<String> filePaths) throws IOException {
153         for (Iterator<String> it = filePaths.iterator(); it.hasNext();) {
154             final String filePath = it.next();
155             if (!FileUtils.isFileInParent(Path.of(filePath).toFile(), parameters.getRootDirFile())) {
156                 it.remove();
157                 messageLog.warning(MessageType.IGNORING_CSS_FILE_OUTSIDE_OF_ROOT_DIR, filePath);
158             }
159         }
160     }

162     /**
163      * Performs processing from the list of file paths for this builder's parameters.
164      *
165      * @param filePaths
166      *            paths of CSS files to process. Non-absolute paths will be taken relative to the current working
167      *            directory. Both platform-specific and '/' as the file separator are supported.
168      *
169      * @throws IOException
170      *             Signals that an I/O exception has occurred.
171      */
172     public void buildSprites(Collection<String> filePaths) throws IOException {
173         final long start = System.currentTimeMillis();
174 
175         final LevelCounterMessageSink levelCounter = new LevelCounterMessageSink();
176         messageLog.addMessageSink(levelCounter);
177 
178         // Collect sprite declarations from all css files
179         final Multimap<String, SpriteImageOccurrence> spriteImageOccurrencesByFile = spriteDirectiveOccurrenceCollector
180                 .collectSpriteImageOccurrences(filePaths);
181 
182         // Merge them, checking for duplicates
183         final Map<String, SpriteImageOccurrence> spriteImageOccurrencesBySpriteId = spriteDirectiveOccurrenceCollector
184                 .mergeSpriteImageOccurrences(spriteImageOccurrencesByFile);
185         final Map<String, SpriteImageDirective> spriteImageDirectivesBySpriteId = new LinkedHashMap<>();
186         for (Map.Entry<String, SpriteImageOccurrence> entry : spriteImageOccurrencesBySpriteId.entrySet()) {
187             spriteImageDirectivesBySpriteId.put(entry.getKey(), entry.getValue().spriteImageDirective);
188         }
189 
190         // Collect sprite references from all css files
191         final Multimap<String, SpriteReferenceOccurrence> spriteEntriesByFile = spriteDirectiveOccurrenceCollector
192                 .collectSpriteReferenceOccurrences(filePaths, spriteImageDirectivesBySpriteId);
193 
194         // Now merge and regroup all files by sprite-id
195         final Multimap<String, SpriteReferenceOccurrence> spriteReferenceOccurrencesBySpriteId = SpriteDirectiveOccurrenceCollector
196                 .mergeSpriteReferenceOccurrences(spriteEntriesByFile);
197 
198         // Build the sprite images
199         messageLog.setCssFile(null);
200         final Multimap<String, SpriteReferenceReplacement> spriteReplacementsByFile = spriteImageBuilder
201                 .buildSpriteImages(spriteImageOccurrencesBySpriteId, spriteReferenceOccurrencesBySpriteId);
202 
203         // Rewrite the CSS
204         rewriteCssFiles(spriteImageOccurrencesByFile, spriteReplacementsByFile);
205 
206         final long stop = System.currentTimeMillis();
207 
208         if (levelCounter.getWarnCount() > 0) {
209             messageLog.status(MessageType.PROCESSING_COMPLETED_WITH_WARNINGS, stop - start,
210                     levelCounter.getWarnCount());
211         } else {
212             messageLog.status(MessageType.PROCESSING_COMPLETED, stop - start);
213         }
214     }

216     /**
217      * Rewrites the original files to refer to the generated sprite images.
218      *
219      * @param spriteImageOccurrencesByFile
220      *            the sprite image occurrences by file
221      * @param spriteReplacementsByFile
222      *            the sprite replacements by file
223      *
224      * @throws IOException
225      *             Signals that an I/O exception has occurred.
226      */
227     private void rewriteCssFiles(final Multimap<String, SpriteImageOccurrence> spriteImageOccurrencesByFile,
228             final Multimap<String, SpriteReferenceReplacement> spriteReplacementsByFile) throws IOException {
229         if (spriteReplacementsByFile.isEmpty()) {
230             // If nothing to replace, still, copy the original file, so that there
231             // is some output file.
232             for (final Map.Entry<String, Collection<SpriteImageOccurrence>> entry : spriteImageOccurrencesByFile.asMap()
233                     .entrySet()) {
234                 final String cssFile = entry.getKey();
235 
236                 createProcessedCss(cssFile, SpriteImageBuilder.getSpriteImageOccurrencesByLineNumber(
237                         spriteImageOccurrencesByFile.get(cssFile)), new HashMap<>());
238             }
239         } else {
240             for (final Map.Entry<String, Collection<SpriteReferenceReplacement>> entry : spriteReplacementsByFile
241                     .asMap().entrySet()) {
242                 final String cssFile = entry.getKey();
243                 final Map<Integer, SpriteReferenceReplacement> spriteReplacementsByLineNumber = SpriteImageBuilder
244                         .getSpriteReplacementsByLineNumber(entry.getValue());
245 
246                 createProcessedCss(cssFile, SpriteImageBuilder.getSpriteImageOccurrencesByLineNumber(
247                         spriteImageOccurrencesByFile.get(cssFile)), spriteReplacementsByLineNumber);
248             }
249         }
250     }

252     /**
253      * Rewrites one CSS file to refer to the generated sprite images.
254      *
255      * @param originalCssFile
256      *            the original css file
257      * @param spriteImageOccurrencesByLineNumber
258      *            the sprite image occurrences by line number
259      * @param spriteReplacementsByLineNumber
260      *            the sprite replacements by line number
261      *
262      * @throws IOException
263      *             Signals that an I/O exception has occurred.
264      */
265     private void createProcessedCss(String originalCssFile,
266             Map<Integer, SpriteImageOccurrence> spriteImageOccurrencesByLineNumber,
267             Map<Integer, SpriteReferenceReplacement> spriteReplacementsByLineNumber) throws IOException {
268         final String processedCssFile = getProcessedCssFile(originalCssFile);
269         messageLog.setCssFile(null);
270         messageLog.info(MessageType.CREATING_CSS_STYLE_SHEET, processedCssFile);
271         messageLog.info(MessageType.READING_CSS, originalCssFile);
272         messageLog.info(MessageType.WRITING_CSS, processedCssFile);
273 
274         String originalCssLine;
275         int originalCssLineNumber = -1;
276         int lastReferenceReplacementLine = -1;
277 
278         boolean markSpriteImages = parameters.isMarkSpriteImages();
279 
280         // Generate UID for sprite file
281         try (BufferedReader originalCssReader = new BufferedReader(
282                 resourceHandler.getResourceAsReader(originalCssFile));
283                 BufferedWriter processedCssWriter = new BufferedWriter(
284                         resourceHandler.getResourceAsWriter(processedCssFile))) {
285             messageLog.setCssFile(originalCssFile);
286 
287             originalCssFile = originalCssFile.replace(File.separatorChar, '/');
288 
289             while ((originalCssLine = originalCssReader.readLine()) != null) {
290                 originalCssLineNumber++;
291                 messageLog.setLine(originalCssLineNumber);
292 
293                 if (originalCssLine.contains("}")) {
294                     lastReferenceReplacementLine = -1;
295                 }
296 
297                 final SpriteImageOccurrence spriteImageOccurrence = spriteImageOccurrencesByLineNumber
298                         .get(originalCssLineNumber);
299                 final SpriteReferenceReplacement spriteReferenceReplacement = spriteReplacementsByLineNumber
300                         .get(originalCssLineNumber);
301 
302                 if (spriteImageOccurrence != null) {
303                     // Ignore line with directive
304                     continue;
305                 }
306 
307                 if (spriteReferenceReplacement != null) {
308                     final boolean important = spriteReferenceReplacement.spriteReferenceOccurrence.important;
309                     lastReferenceReplacementLine = originalCssLineNumber;
310 
311                     processedCssWriter.write("  background-image: url('"
312                             + getRelativeToReplacementLocation(spriteReferenceReplacement.spriteImage.resolvedPath,
313                                     originalCssFile, spriteReferenceReplacement)
314                             + "')" + (important ? " !important" : "") + ";"
315                             + (markSpriteImages ? " /** sprite:sprite */" : "") + "\n");
316 
317                     processedCssWriter
318                             .write("  background-position: " + spriteReferenceReplacement.horizontalPositionString + " "
319                                     + spriteReferenceReplacement.verticalPositionString
320                                     + (important ? " !important" : "") + ";\n");
321 
322                     // If the sprite scale is not 1, write out a background-size directive
323                     final float scale = spriteReferenceReplacement.spriteImage.scaleRatio;
324                     if (scale != 1.0f) {
325                         processedCssWriter.write("  background-size: "
326                                 + Math.round(spriteReferenceReplacement.spriteImage.spriteWidth / scale) + "px "
327                                 + Math.round(spriteReferenceReplacement.spriteImage.spriteHeight / scale) + "px;\n");
328                     }
329 
330                     continue;
331                 }
332 
333                 if (lastReferenceReplacementLine >= 0) {
334                     for (final String property : OVERRIDING_PROPERTIES) {
335                         if (originalCssLine.contains(property)) {
336                             messageLog.warning(MessageType.OVERRIDING_PROPERTY_FOUND, property,
337                                     lastReferenceReplacementLine);
338                         }
339                     }
340                 }
341 
342                 // Just write the original line
343                 processedCssWriter.write(originalCssLine + "\n");
344             }
345 
346             messageLog.setCssFile(null);
347         }
348     }

350     /**
351      * Returns the sprite image's imagePath relative to the CSS in which we're making replacements. The imagePath is
352      * relative to the CSS which declared the sprite image. As it may happen that the image is referenced in another CSS
353      * file, we must make sure the paths are correctly translated.
354      *
355      * @param imagePath
356      *            the image path
357      * @param originalCssFile
358      *            the original css file
359      * @param spriteReferenceReplacement
360      *            the sprite reference replacement
361      *
362      * @return the relative to replacement location
363      */
364     private String getRelativeToReplacementLocation(String imagePath, String originalCssFile,
365             final SpriteReferenceReplacement spriteReferenceReplacement) {
366         final String declaringCssPath = spriteReferenceReplacement.spriteImage.spriteImageOccurrence.cssFile
367                 .replace(File.separatorChar, '/');
368         final String declarationReplacementRelativePath = PathUtils
369                 .getRelativeFilePath(originalCssFile.substring(0, originalCssFile.lastIndexOf('/')),
370                         declaringCssPath.substring(0, declaringCssPath.lastIndexOf('/')))
371                 .replace(File.separatorChar, '/');
372         return FileUtils.canonicalize(
373                 (Strings.isNullOrEmpty(declarationReplacementRelativePath) || originalCssFile.equals(declaringCssPath)
374                         ? ""
375                         : declarationReplacementRelativePath + '/') + imagePath,
376                 "/");
377     }

379     /**
380      * Gets the name of the processed CSS file.
381      *
382      * @param originalCssFile
383      *            the original css file
384      *
385      * @return the processed css file
386      */
387     String getProcessedCssFile(String originalCssFile) {
388         final int lastDotIndex = originalCssFile.lastIndexOf('.');
389         final String processedCssFile;
390         if (lastDotIndex >= 0) {
391             processedCssFile = originalCssFile.substring(0, lastDotIndex) + parameters.getCssFileSuffix()
392                     + originalCssFile.substring(lastDotIndex);
393         } else {
394             processedCssFile = originalCssFile + parameters.getCssFileSuffix();
395         }
396 
397         if (parameters.hasOutputDir()) {
398             return FileUtils.changeRoot(processedCssFile, parameters.getRootDir(), parameters.getOutputDir());
399         }
400         return processedCssFile;
401     }
