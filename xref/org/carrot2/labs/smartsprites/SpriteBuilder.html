104     /**
105      * Creates a {@link SpriteBuilder} with the provided parameters and log.
106      *
107      * @param parameters
108      *            the parameters
109      * @param messageLog
110      *            the message log
111      * @param resourceHandler
112      *            the resource handler
113      */
114     public SpriteBuilder(SmartSpritesParameters parameters, MessageLog messageLog, ResourceHandler resourceHandler) {
115         this.messageLog = messageLog;
116         this.parameters = parameters;
117         this.resourceHandler = resourceHandler;
118         spriteDirectiveOccurrenceCollector = new SpriteDirectiveOccurrenceCollector(messageLog, resourceHandler);
119         spriteImageBuilder = new SpriteImageBuilder(parameters, messageLog, resourceHandler);
120     }

122     /**
123      * Performs processing for this builder's parameters. This method resolves all paths against the local file system.
124      *
125      * @throws IOException
126      *             Signals that an I/O exception has occurred.
127      */
128     public void buildSprites() throws IOException {
129         if (!parameters.validate(messageLog)) {
130             return;
131         }
132 
133         final Collection<String> filePaths;
134         if (parameters.getCssFiles() != null && !parameters.getCssFiles().isEmpty()) {
135             // Take directly provided css fle paths
136             filePaths = parameters.getCssFiles();
137 
138             // If root dir is provided, filter out those files that are outside root dir
139             if (StringUtils.isNotBlank(parameters.getOutputDir())) {
140                 filterFilesOutsideRootDir(filePaths);
141             }
142 
143             // Make sure the files exist and are really files
144             for (Iterator<String> it = filePaths.iterator(); it.hasNext();) {
145                 final String path = it.next();
146                 final File file = new File(path);
147                 if (file.exists()) {
148                     if (!file.isFile()) {
149                         messageLog.warning(MessageType.CSS_PATH_IS_NOT_A_FILE, path);
150                         it.remove();
151                     }
152                 } else {
153                     messageLog.warning(MessageType.CSS_FILE_DOES_NOT_EXIST, path);
154                     it.remove();
155                 }
156             }
157         } else {
158             // Take all css files from the root dir
159             final List<File> files = Lists.newArrayList(org.apache.commons.io.FileUtils
160                     .listFiles(parameters.getRootDirFile(), new String[] { "css" }, true));
161             Collections.sort(files, Comparator.comparing(File::getAbsolutePath));
162 
163             filePaths = new ArrayList<>();
164             for (File file : files) {
165                 filePaths.add(file.getPath());
166             }
167         }
168 
169         buildSprites(filePaths);
170     }
171 
172     /**
173      * Filter files outside root dir.
174      *
175      * @param filePaths
176      *            the file paths
177      *
178      * @throws IOException
179      *             Signals that an I/O exception has occurred.
180      */
181     private void filterFilesOutsideRootDir(Collection<String> filePaths) throws IOException {
182         for (Iterator<String> it = filePaths.iterator(); it.hasNext();) {
183             final String filePath = it.next();
184             if (!FileUtils.isFileInParent(new File(filePath), parameters.getRootDirFile())) {
185                 it.remove();
186                 messageLog.warning(MessageType.IGNORING_CSS_FILE_OUTSIDE_OF_ROOT_DIR, filePath);
187             }
188         }
189     }

191     /**
192      * Performs processing from the list of file paths for this builder's parameters.
193      *
194      * @param filePaths
195      *            paths of CSS files to process. Non-absolute paths will be taken relative to the current working
196      *            directory. Both platform-specific and '/' as the file separator are supported.
197      *
198      * @throws IOException
199      *             Signals that an I/O exception has occurred.
200      */
201     public void buildSprites(Collection<String> filePaths) throws IOException {
202         final long start = System.currentTimeMillis();
203 
204         final LevelCounterMessageSink levelCounter = new LevelCounterMessageSink();
205         messageLog.addMessageSink(levelCounter);
206 
207         // Collect sprite declarations from all css files
208         final Multimap<String, SpriteImageOccurrence> spriteImageOccurrencesByFile = spriteDirectiveOccurrenceCollector
209                 .collectSpriteImageOccurrences(filePaths);
210 
211         // Merge them, checking for duplicates
212         final Map<String, SpriteImageOccurrence> spriteImageOccurrencesBySpriteId = spriteDirectiveOccurrenceCollector
213                 .mergeSpriteImageOccurrences(spriteImageOccurrencesByFile);
214         final Map<String, SpriteImageDirective> spriteImageDirectivesBySpriteId = new LinkedHashMap<>();
215         for (Map.Entry<String, SpriteImageOccurrence> entry : spriteImageOccurrencesBySpriteId.entrySet()) {
216             spriteImageDirectivesBySpriteId.put(entry.getKey(), entry.getValue().spriteImageDirective);
217         }
218 
219         // Collect sprite references from all css files
220         final Multimap<String, SpriteReferenceOccurrence> spriteEntriesByFile = spriteDirectiveOccurrenceCollector
221                 .collectSpriteReferenceOccurrences(filePaths, spriteImageDirectivesBySpriteId);
222 
223         // Now merge and regroup all files by sprite-id
224         final Multimap<String, SpriteReferenceOccurrence> spriteReferenceOccurrencesBySpriteId = SpriteDirectiveOccurrenceCollector
225                 .mergeSpriteReferenceOccurrences(spriteEntriesByFile);
226 
227         // Build the sprite images
228         messageLog.setCssFile(null);
229         final Multimap<String, SpriteReferenceReplacement> spriteReplacementsByFile = spriteImageBuilder
230                 .buildSpriteImages(spriteImageOccurrencesBySpriteId, spriteReferenceOccurrencesBySpriteId);
231 
232         // Rewrite the CSS
233         rewriteCssFiles(spriteImageOccurrencesByFile, spriteReplacementsByFile);
234 
235         final long stop = System.currentTimeMillis();
236 
237         if (levelCounter.getWarnCount() > 0) {
238             messageLog.status(MessageType.PROCESSING_COMPLETED_WITH_WARNINGS, stop - start,
239                     levelCounter.getWarnCount());
240         } else {
241             messageLog.status(MessageType.PROCESSING_COMPLETED, stop - start);
242         }
243     }

245     /**
246      * Rewrites the original files to refer to the generated sprite images.
247      *
248      * @param spriteImageOccurrencesByFile
249      *            the sprite image occurrences by file
250      * @param spriteReplacementsByFile
251      *            the sprite replacements by file
252      *
253      * @throws IOException
254      *             Signals that an I/O exception has occurred.
255      */
256     private void rewriteCssFiles(final Multimap<String, SpriteImageOccurrence> spriteImageOccurrencesByFile,
257             final Multimap<String, SpriteReferenceReplacement> spriteReplacementsByFile) throws IOException {
258         if (spriteReplacementsByFile.isEmpty()) {
259             // If nothing to replace, still, copy the original file, so that there
260             // is some output file.
261             for (final Map.Entry<String, Collection<SpriteImageOccurrence>> entry : spriteImageOccurrencesByFile.asMap()
262                     .entrySet()) {
263                 final String cssFile = entry.getKey();
264 
265                 createProcessedCss(cssFile, SpriteImageBuilder.getSpriteImageOccurrencesByLineNumber(
266                         spriteImageOccurrencesByFile.get(cssFile)), new HashMap<>());
267             }
268         } else {
269             for (final Map.Entry<String, Collection<SpriteReferenceReplacement>> entry : spriteReplacementsByFile
270                     .asMap().entrySet()) {
271                 final String cssFile = entry.getKey();
272                 final Map<Integer, SpriteReferenceReplacement> spriteReplacementsByLineNumber = SpriteImageBuilder
273                         .getSpriteReplacementsByLineNumber(entry.getValue());
274 
275                 createProcessedCss(cssFile, SpriteImageBuilder.getSpriteImageOccurrencesByLineNumber(
276                         spriteImageOccurrencesByFile.get(cssFile)), spriteReplacementsByLineNumber);
277             }
278         }
279     }

281     /**
282      * Rewrites one CSS file to refer to the generated sprite images.
283      *
284      * @param originalCssFile
285      *            the original css file
286      * @param spriteImageOccurrencesByLineNumber
287      *            the sprite image occurrences by line number
288      * @param spriteReplacementsByLineNumber
289      *            the sprite replacements by line number
290      *
291      * @throws IOException
292      *             Signals that an I/O exception has occurred.
293      */
294     private void createProcessedCss(String originalCssFile,
295             Map<Integer, SpriteImageOccurrence> spriteImageOccurrencesByLineNumber,
296             Map<Integer, SpriteReferenceReplacement> spriteReplacementsByLineNumber) throws IOException {
297         final String processedCssFile = getProcessedCssFile(originalCssFile);
298         messageLog.setCssFile(null);
299         messageLog.info(MessageType.CREATING_CSS_STYLE_SHEET, processedCssFile);
300         messageLog.info(MessageType.READING_CSS, originalCssFile);
301         messageLog.info(MessageType.WRITING_CSS, processedCssFile);
302 
303         String originalCssLine;
304         int originalCssLineNumber = -1;
305         int lastReferenceReplacementLine = -1;
306 
307         boolean markSpriteImages = parameters.isMarkSpriteImages();
308 
309         // Generate UID for sprite file
310         try (BufferedReader originalCssReader = new BufferedReader(
311                 resourceHandler.getResourceAsReader(originalCssFile));
312                 BufferedWriter processedCssWriter = new BufferedWriter(
313                         resourceHandler.getResourceAsWriter(processedCssFile))) {
314             messageLog.setCssFile(originalCssFile);
315 
316             originalCssFile = originalCssFile.replace(File.separatorChar, '/');
317 
318             while ((originalCssLine = originalCssReader.readLine()) != null) {
319                 originalCssLineNumber++;
320                 messageLog.setLine(originalCssLineNumber);
321 
322                 if (originalCssLine.contains("}")) {
323                     lastReferenceReplacementLine = -1;
324                 }
325 
326                 final SpriteImageOccurrence spriteImageOccurrence = spriteImageOccurrencesByLineNumber
327                         .get(originalCssLineNumber);
328                 final SpriteReferenceReplacement spriteReferenceReplacement = spriteReplacementsByLineNumber
329                         .get(originalCssLineNumber);
330 
331                 if (spriteImageOccurrence != null) {
332                     // Ignore line with directive
333                     continue;
334                 }
335 
336                 if (spriteReferenceReplacement != null) {
337                     final boolean important = spriteReferenceReplacement.spriteReferenceOccurrence.important;
338                     lastReferenceReplacementLine = originalCssLineNumber;
339 
340                     processedCssWriter.write("  background-image: url('"
341                             + getRelativeToReplacementLocation(spriteReferenceReplacement.spriteImage.resolvedPath,
342                                     originalCssFile, spriteReferenceReplacement)
343                             + "')" + (important ? " !important" : "") + ";"
344                             + (markSpriteImages ? " /** sprite:sprite */" : "") + "\n");
345 
346                     processedCssWriter
347                             .write("  background-position: " + spriteReferenceReplacement.horizontalPositionString + " "
348                                     + spriteReferenceReplacement.verticalPositionString
349                                     + (important ? " !important" : "") + ";\n");
350 
351                     // If the sprite scale is not 1, write out a background-size directive
352                     final float scale = spriteReferenceReplacement.spriteImage.scaleRatio;
353                     if (scale != 1.0f) {
354                         processedCssWriter.write("  background-size: "
355                                 + Math.round(spriteReferenceReplacement.spriteImage.spriteWidth / scale) + "px "
356                                 + Math.round(spriteReferenceReplacement.spriteImage.spriteHeight / scale) + "px;\n");
357                     }
358 
359                     continue;
360                 }
361 
362                 if (lastReferenceReplacementLine >= 0) {
363                     for (final String property : OVERRIDING_PROPERTIES) {
364                         if (originalCssLine.contains(property)) {
365                             messageLog.warning(MessageType.OVERRIDING_PROPERTY_FOUND, property,
366                                     lastReferenceReplacementLine);
367                         }
368                     }
369                 }
370 
371                 // Just write the original line
372                 processedCssWriter.write(originalCssLine + "\n");
373             }
374 
375             messageLog.setCssFile(null);
376         }
377     }

379     /**
380      * Returns the sprite image's imagePath relative to the CSS in which we're making replacements. The imagePath is
381      * relative to the CSS which declared the sprite image. As it may happen that the image is referenced in another CSS
382      * file, we must make sure the paths are correctly translated.
383      *
384      * @param imagePath
385      *            the image path
386      * @param originalCssFile
387      *            the original css file
388      * @param spriteReferenceReplacement
389      *            the sprite reference replacement
390      *
391      * @return the relative to replacement location
392      */
393     private String getRelativeToReplacementLocation(String imagePath, String originalCssFile,
394             final SpriteReferenceReplacement spriteReferenceReplacement) {
395         final String declaringCssPath = spriteReferenceReplacement.spriteImage.spriteImageOccurrence.cssFile
396                 .replace(File.separatorChar, '/');
397         final String declarationReplacementRelativePath = PathUtils
398                 .getRelativeFilePath(originalCssFile.substring(0, originalCssFile.lastIndexOf('/')),
399                         declaringCssPath.substring(0, declaringCssPath.lastIndexOf('/')))
400                 .replace(File.separatorChar, '/');
401         return FileUtils.canonicalize(
402                 (Strings.isNullOrEmpty(declarationReplacementRelativePath) || originalCssFile.equals(declaringCssPath)
403                         ? "" : declarationReplacementRelativePath + '/') + imagePath,
404                 "/");
405     }

407     /**
408      * Gets the name of the processed CSS file.
409      *
410      * @param originalCssFile
411      *            the original css file
412      *
413      * @return the processed css file
414      */
415     String getProcessedCssFile(String originalCssFile) {
416         final int lastDotIndex = originalCssFile.lastIndexOf('.');
417         final String processedCssFile;
418         if (lastDotIndex >= 0) {
419             processedCssFile = originalCssFile.substring(0, lastDotIndex) + parameters.getCssFileSuffix()
420                     + originalCssFile.substring(lastDotIndex);
421         } else {
422             processedCssFile = originalCssFile + parameters.getCssFileSuffix();
423         }
424 
425         if (parameters.hasOutputDir()) {
426             return FileUtils.changeRoot(processedCssFile, parameters.getRootDir(), parameters.getOutputDir());
427         }
428         return processedCssFile;
429     }
