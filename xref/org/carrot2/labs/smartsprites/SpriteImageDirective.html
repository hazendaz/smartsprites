85      /**
86       * Defines the layout of this sprite.
87       */
88      public enum SpriteImageLayout {
89          /**
90           * Vertical layout, images stacked on each other.
91           */
92          VERTICAL,
93  
94          /**
95           * Horizontal layout, images next to each other.
96           */
97          HORIZONTAL;
98  
99          /** The value. */
100         private final String value;
101 
102         /**
103          * Instantiates a new sprite image layout.
104          */
105         SpriteImageLayout() {
106             this.value = name().toLowerCase(Locale.ENGLISH);
107         }
108 
109         @Override
110         public String toString() {
111             return value;
112         }
113 
114         /**
115          * Values as string.
116          *
117          * @return the string
118          */
119         public static String valuesAsString() {
120             final String list = Lists.newArrayList(values()).toString();
121             return list.substring(1, list.length() - 1);
122         }
123     }

265     /**
266      * Pattern for a simple syntactic check of the image path.
267      */
268     private static final Pattern IMAGE_PATH_PATTERN = Pattern.compile("([^${}]*|\\$\\{[^}]*\\})*");
269 
270     /**
271      * Pattern for extracting variables from image path.
272      */
273     private static final Pattern IMAGE_PATH_VARIABLE_PATTERN = Pattern.compile("\\$\\{([a-z]*)\\}");
274 
275     /**
276      * Variable names allowed in image path.
277      */
278     private static final Set<String> ALLOWED_VARIABLES = ImmutableSet.of(PROPERTY_SPRITE_ID,
279             SpriteUidType.DATE.toString(), SpriteUidType.SHA512.toString());

336     /**
337      * Parses a string into a {@link SpriteImageDirective}, logging messages to the provided {@link MessageLog}s.
338      *
339      * @param directiveString
340      *            the directive string
341      * @param messageCollector
342      *            the message collector
343      *
344      * @return the sprite image directive
345      */
346     public static SpriteImageDirective parse(String directiveString, MessageLog messageCollector) {
347         final Map<String, CssProperty> rules = CssSyntaxUtils
348                 .propertiesAsMap(CssSyntaxUtils.extractRules(directiveString, messageCollector));
349 
350         final Set<String> properties = Sets.newLinkedHashSet(rules.keySet());
351         properties.removeAll(ALLOWED_PROPERTIES);
352         properties.removeAll(SpriteLayoutProperties.ALLOWED_PROPERTIES);
353         if (!properties.isEmpty()) {
354             messageCollector.warning(MessageType.UNSUPPORTED_PROPERTIES_FOUND, CollectionUtils.toString(properties));
355         }
356 
357         if (!CssSyntaxUtils.hasNonBlankValue(rules, PROPERTY_SPRITE_ID)) {
358             messageCollector.warning(MessageType.SPRITE_ID_NOT_FOUND);
359             return null;
360         }
361 
362         if (!CssSyntaxUtils.hasNonBlankValue(rules, PROPERTY_SPRITE_IMAGE_URL)) {
363             messageCollector.warning(MessageType.SPRITE_IMAGE_URL_NOT_FOUND);
364             return null;
365         }
366 
367         final String id = rules.get(PROPERTY_SPRITE_ID).value;
368 
369         final SpriteUidType uidGenerator = valueOf(CssSyntaxUtils.getValue(rules, PROPERTY_SPRITE_IMAGE_UID_SUFFIX),
370                 SpriteUidType.class, SpriteUidType.NONE, messageCollector, MessageType.UNSUPPORTED_UID_TYPE);
371         if (uidGenerator != SpriteUidType.NONE) {
372             messageCollector.deprecation(MessageType.DEPRECATED_SPRITE_IMAGE_UID, uidGenerator.toString());
373         }
374 
375         // Image path. If the path does not match a regular expression, issue a warning.
376         final String imagePath = CssSyntaxUtils.unpackUrl(rules.get(PROPERTY_SPRITE_IMAGE_URL).value);
377         if (IMAGE_PATH_PATTERN.matcher(imagePath).matches()) {
378             // Check variable names
379             final Matcher variableMatcher = IMAGE_PATH_VARIABLE_PATTERN.matcher(imagePath);
380             while (variableMatcher.find()) {
381                 if (variableMatcher.groupCount() == 1 && !ALLOWED_VARIABLES.contains(variableMatcher.group(1))) {
382                     messageCollector.warning(MessageType.UNSUPPORTED_VARIABLE_IN_SPRITE_IMAGE_PATH,
383                             variableMatcher.group(1));
384                 }
385             }
386         } else {
387             // Just issue a warning
388             messageCollector.warning(MessageType.MALFORMED_SPRITE_IMAGE_PATH, imagePath);
389         }
390 
391         // Layout is optional
392         final SpriteImageLayout layout = valueOf(CssSyntaxUtils.getValue(rules, PROPERTY_SPRITE_IMAGE_LAYOUT),
393                 SpriteImageLayout.class, SpriteImageLayout.VERTICAL, messageCollector, MessageType.UNSUPPORTED_LAYOUT);
394 
395         // Infer format from image path
396         SpriteImageFormat format;
397         final int lastDotIndex = imagePath.lastIndexOf('.');
398         if (lastDotIndex < 0 || lastDotIndex == imagePath.length() - 1) {
399             messageCollector.warning(MessageType.CANNOT_DETERMINE_IMAGE_FORMAT, imagePath);
400             format = SpriteImageFormat.PNG;
401         } else {
402             final int questionMarkIndex = imagePath.indexOf('?', lastDotIndex);
403             final String formatValue = questionMarkIndex >= 0 ? imagePath.substring(lastDotIndex + 1, questionMarkIndex)
404                     : imagePath.substring(lastDotIndex + 1);
405             try {
406                 format = SpriteImageFormat.getValue(formatValue);
407             } catch (final IllegalArgumentException e) {
408                 messageCollector.warning(MessageType.UNSUPPORTED_SPRITE_IMAGE_FORMAT, formatValue);
409                 format = SpriteImageFormat.PNG;
410             }
411         }
412 
413         // Matte color
414         final Color matteColor;
415         if (CssSyntaxUtils.hasNonBlankValue(rules, PROPERTY_SPRITE_MATTE_COLOR)) {
416             matteColor = CssSyntaxUtils.parseColor(rules.get(PROPERTY_SPRITE_MATTE_COLOR).value, messageCollector,
417                     null);
418         } else {
419             matteColor = null;
420         }
421 
422         final float scale;
423         if (CssSyntaxUtils.hasNonBlankValue(rules, PROPERTY_SPRITE_SCALE)) {
424             scale = Float.parseFloat(rules.get(PROPERTY_SPRITE_SCALE).value);
425         } else {
426             scale = 1.0f;
427         }
428 
429         return new SpriteImageDirective(id, imagePath, layout, format, matteColor, uidGenerator, scale,
430                 SpriteLayoutProperties.parse(directiveString, layout, messageCollector));
431     }

433     /**
434      * Value of.
435      *
436      * @param <T>
437      *            the generic type
438      * @param stringValue
439      *            the string value
440      * @param enumClass
441      *            the enum class
442      * @param defaultValue
443      *            the default value
444      * @param messageCollector
445      *            the message collector
446      * @param messageType
447      *            the message type
448      *
449      * @return the t
450      */
451     private static <T extends Enum<T>> T valueOf(String stringValue, Class<T> enumClass, T defaultValue,
452             MessageLog messageCollector, MessageType messageType) {
453         if (!StringUtils.isNotBlank(stringValue)) {
454             return defaultValue;
455         }
456         try {
457             return Enum.valueOf(enumClass, stringValue.toUpperCase(Locale.ENGLISH));
458         } catch (IllegalArgumentException e) {
459             messageCollector.warning(messageType, stringValue);
460             return defaultValue;
461         }
462     }
