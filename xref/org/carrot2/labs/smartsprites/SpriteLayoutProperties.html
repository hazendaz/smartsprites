50      /**
51       * Alignment of the individual image within the sprite image.
52       */
53      public enum SpriteAlignment {
54          /**
55           * To the left edge of a vertical sprite.
56           */
57          LEFT,
58  
59          /**
60           * To the right edge of a vertical sprite.
61           */
62          RIGHT,
63  
64          /**
65           * To the top edge of a horizontal sprite.
66           */
67          TOP,
68  
69          /**
70           * To the bottom edge of a horizontal sprite.
71           */
72          BOTTOM,
73  
74          /**
75           * Repeated across the full width/ height of the sprite image.
76           */
77          REPEAT,
78  
79          /** To the center of a vertical or horizontal sprite. */
80          CENTER;
81  
82          /** The value. */
83          private final String value;
84  
85          /**
86           * Instantiates a new sprite alignment.
87           */
88          SpriteAlignment() {
89              this.value = name().toLowerCase(Locale.ENGLISH);
90          }
91  
92          @Override
93          public String toString() {
94              return value;
95          }
96  
97          /**
98           * Gets the value.
99           *
100          * @param value
101          *            the value
102          *
103          * @return the value
104          */
105         public static SpriteAlignment getValue(String value) {
106             return valueOf(value.toUpperCase(Locale.ENGLISH));
107         }
108 
109         /**
110          * Values as string.
111          *
112          * @return the string
113          */
114         public static String valuesAsString() {
115             final String list = new ArrayList<>(Arrays.asList(values())).toString();
116             return list.substring(1, list.length() - 1);
117         }
118     }

168     /**
169      * Parses a {@link SpriteLayoutProperties} from the provided {@link String} logging messages to the provided
170      * {@link MessageLog}.
171      *
172      * @param directiveString
173      *            the directive string
174      * @param spriteImageLayout
175      *            the sprite image layout
176      * @param messageCollector
177      *            the message collector

179      * @return the sprite layout properties

187     /**
188      * Parses a {@link SpriteLayoutProperties} from the provided {@link String}, using the provided defaults and logging
189      * messages to the provided {@link MessageLog}.
190      *
191      * @param directiveString
192      *            the directive string
193      * @param spriteImageLayout
194      *            the sprite image layout
195      * @param defaults
196      *            the defaults
197      * @param messageCollector
198      *            the message collector
199      *
200      * @return the sprite layout properties
201      */
202     public static SpriteLayoutProperties parse(String directiveString, SpriteImageLayout spriteImageLayout,
203             SpriteLayoutProperties defaults, MessageLog messageCollector) {
204         final Map<String, CssProperty> rules = CssSyntaxUtils
205                 .propertiesAsMap(CssSyntaxUtils.extractRules(directiveString, messageCollector));
206 
207         // We don't check for allowed properties here. The check, including
208         // sprite layout properties will be done when parsing the directive
209         // that embeds sprite layout properties.
210 
211         // Alignment is optional
212         SpriteAlignment alignment;
213         if (CssSyntaxUtils.hasNonBlankValue(rules, PROPERTY_SPRITE_ALIGNMENT)) {
214             final String alignmentValue = rules.get(PROPERTY_SPRITE_ALIGNMENT).value;
215             try {
216                 alignment = correctAlignment(spriteImageLayout, SpriteAlignment.getValue(alignmentValue),
217                         messageCollector);
218             } catch (final IllegalArgumentException e) {
219                 messageCollector.warning(MessageType.UNSUPPORTED_ALIGNMENT, alignmentValue);
220                 alignment = getDefaultAlignment(spriteImageLayout);
221             }
222         } else {
223             alignment = defaults.alignment;
224         }
225 
226         // Parse margins
227         final int marginLeft = getMargin(PROPERTY_SPRITE_MARGIN_LEFT, rules, defaults.marginLeft, messageCollector);
228         final int marginRight = getMargin(PROPERTY_SPRITE_MARGIN_RIGHT, rules, defaults.marginRight, messageCollector);
229         final int marginTop = getMargin(PROPERTY_SPRITE_MARGIN_TOP, rules, defaults.marginTop, messageCollector);
230         final int marginBottom = getMargin(PROPERTY_SPRITE_MARGIN_BOTTOM, rules, defaults.marginBottom,
231                 messageCollector);
232 
233         return new SpriteLayoutProperties(alignment, marginLeft, marginRight, marginTop, marginBottom);
234     }

236     /**
237      * Corrects sprite alignment if necessary based on the layout of the enclosing sprite image.
238      *
239      * @param spriteImageLayout
240      *            the sprite image layout
241      * @param alignment
242      *            the alignment
243      * @param messageCollector
244      *            the message collector
245      *
246      * @return the sprite alignment
247      */
248     private static SpriteAlignment correctAlignment(SpriteImageLayout spriteImageLayout, SpriteAlignment alignment,
249             MessageLog messageCollector) {
250         if (spriteImageLayout.equals(SpriteImageLayout.HORIZONTAL)) {
251             if (alignment.equals(SpriteAlignment.LEFT) || alignment.equals(SpriteAlignment.RIGHT)) {
252                 messageCollector.warning(MessageType.ONLY_TOP_OR_BOTTOM_ALIGNMENT_ALLOWED, alignment.value);
253                 return SpriteAlignment.TOP;
254             }
255         } else if (alignment.equals(SpriteAlignment.TOP) || alignment.equals(SpriteAlignment.BOTTOM)) {
256             messageCollector.warning(MessageType.ONLY_LEFT_OR_RIGHT_ALIGNMENT_ALLOWED, alignment.value);
257             return SpriteAlignment.LEFT;
258         }
259 
260         return alignment;
261     }
262 
263     /**
264      * Returns default alignment for given sprite image directive.
265      *
266      * @param spriteImageLayout
267      *            the sprite image layout
268      *
269      * @return the default alignment
270      */
271     private static SpriteAlignment getDefaultAlignment(SpriteImageLayout spriteImageLayout) {
272         if (spriteImageLayout.equals(SpriteImageLayout.HORIZONTAL)) {
273             return SpriteAlignment.TOP;
274         }
275         return SpriteAlignment.LEFT;
276     }

278     /**
279      * Parses margin value.
280      *
281      * @param marginRule
282      *            the margin rule
283      * @param rules
284      *            the rules
285      * @param defaultMargin
286      *            the default margin
287      * @param messageLog
288      *            the message log
289      *
290      * @return the margin
291      */
292     private static int getMargin(String marginRule, Map<String, CssProperty> rules, int defaultMargin,
293             MessageLog messageLog) {
294         if (!CssSyntaxUtils.hasNonBlankValue(rules, marginRule)) {
295             return defaultMargin;
296         }
297         final String rawMarginValue = rules.get(marginRule).value;
298         String marginValue = rawMarginValue;
299         if (marginValue.toLowerCase(Locale.ENGLISH).endsWith("px")) {
300             marginValue = marginValue.substring(0, marginValue.length() - 2);
301         }
302         try {
303             int marginIntValue = Integer.parseInt(marginValue);
304             if (marginIntValue < 0) {
305                 messageLog.warning(MessageType.IGNORING_NEGATIVE_MARGIN_VALUE, marginRule);
306                 marginIntValue = 0;
307             }
308 
309             return marginIntValue;
310         } catch (final NumberFormatException e) {
311             messageLog.warning(MessageType.CANNOT_PARSE_MARGIN_VALUE, rawMarginValue);
312             return 0;
313         }
314     }
