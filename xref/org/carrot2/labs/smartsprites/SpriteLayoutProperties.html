53   * Represents common sprite layout properties that can be used both in {@link SpriteImageDirective} and

67      /** The Constant PROPERTY_SPRITE_MARGIN_RIGHT. */
68      public static final String PROPERTY_SPRITE_MARGIN_RIGHT = "sprite-margin-right";

70      /** The Constant PROPERTY_SPRITE_MARGIN_LEFT. */
71      public static final String PROPERTY_SPRITE_MARGIN_LEFT = "sprite-margin-left";

73      /** Allowed properties of this directive. */
74      static final Set<String> ALLOWED_PROPERTIES = ImmutableSet.of(PROPERTY_SPRITE_ALIGNMENT,
75              PROPERTY_SPRITE_MARGIN_LEFT, PROPERTY_SPRITE_MARGIN_RIGHT, PROPERTY_SPRITE_MARGIN_TOP,
76              PROPERTY_SPRITE_MARGIN_BOTTOM);

78      /**
79       * Alignment of the individual image within the sprite image.
80       */
81      public enum SpriteAlignment {
82          /**
83           * To the left edge of a vertical sprite.
84           */
85          LEFT,
86  
87          /**
88           * To the right edge of a vertical sprite.
89           */
90          RIGHT,
91  
92          /**
93           * To the top edge of a horizontal sprite.
94           */
95          TOP,
96  
97          /**
98           * To the bottom edge of a horizontal sprite.
99           */
100         BOTTOM,
101 
102         /**
103          * Repeated across the full width/ height of the sprite image.
104          */
105         REPEAT,
106 
107         /** To the center of a vertical or horizontal sprite. */
108         CENTER;
109 
110         /** The value. */
111         private final String value;
112 
113         /**
114          * Instantiates a new sprite alignment.
115          */
116         SpriteAlignment() {
117             this.value = name().toLowerCase(Locale.ENGLISH);
118         }
119 
120         @Override
121         public String toString() {
122             return value;
123         }
124 
125         /**
126          * Gets the value.
127          *
128          * @param value
129          *            the value
130          *
131          * @return the value
132          */
133         public static SpriteAlignment getValue(String value) {
134             return valueOf(value.toUpperCase(Locale.ENGLISH));
135         }
136 
137         /**
138          * Values as string.
139          *
140          * @return the string
141          */
142         public static String valuesAsString() {
143             final String list = Lists.newArrayList(values()).toString();
144             return list.substring(1, list.length() - 1);
145         }
146     }

196     /**
197      * Parses a {@link SpriteLayoutProperties} from the provided {@link String} logging messages to the provided
198      * {@link MessageLog}.
199      *
200      * @param directiveString
201      *            the directive string
202      * @param spriteImageLayout
203      *            the sprite image layout
204      * @param messageCollector
205      *            the message collector
206      *
207      * @return the sprite layout properties
208      */
209     public static SpriteLayoutProperties parse(String directiveString, SpriteImageLayout spriteImageLayout,
210             MessageLog messageCollector) {
211         return parse(directiveString, spriteImageLayout, new SpriteLayoutProperties(spriteImageLayout),
212                 messageCollector);
213     }
214 
215     /**
216      * Parses a {@link SpriteLayoutProperties} from the provided {@link String}, using the provided defaults and logging
217      * messages to the provided {@link MessageLog}.
218      *
219      * @param directiveString
220      *            the directive string
221      * @param spriteImageLayout
222      *            the sprite image layout
223      * @param defaults
224      *            the defaults
225      * @param messageCollector
226      *            the message collector
227      *
228      * @return the sprite layout properties
229      */
230     public static SpriteLayoutProperties parse(String directiveString, SpriteImageLayout spriteImageLayout,
231             SpriteLayoutProperties defaults, MessageLog messageCollector) {
232         final Map<String, CssProperty> rules = CssSyntaxUtils
233                 .propertiesAsMap(CssSyntaxUtils.extractRules(directiveString, messageCollector));
234 
235         // We don't check for allowed properties here. The check, including
236         // sprite layout properties will be done when parsing the directive
237         // that embeds sprite layout properties.
238 
239         // Alignment is optional
240         SpriteAlignment alignment;
241         if (CssSyntaxUtils.hasNonBlankValue(rules, PROPERTY_SPRITE_ALIGNMENT)) {
242             final String alignmentValue = rules.get(PROPERTY_SPRITE_ALIGNMENT).value;
243             try {
244                 alignment = correctAlignment(spriteImageLayout, SpriteAlignment.getValue(alignmentValue),
245                         messageCollector);
246             } catch (final IllegalArgumentException e) {
247                 messageCollector.warning(MessageType.UNSUPPORTED_ALIGNMENT, alignmentValue);
248                 alignment = getDefaultAlignment(spriteImageLayout);
249             }
250         } else {
251             alignment = defaults.alignment;
252         }
253 
254         // Parse margins
255         final int marginLeft = getMargin(PROPERTY_SPRITE_MARGIN_LEFT, rules, defaults.marginLeft, messageCollector);
256         final int marginRight = getMargin(PROPERTY_SPRITE_MARGIN_RIGHT, rules, defaults.marginRight, messageCollector);
257         final int marginTop = getMargin(PROPERTY_SPRITE_MARGIN_TOP, rules, defaults.marginTop, messageCollector);
258         final int marginBottom = getMargin(PROPERTY_SPRITE_MARGIN_BOTTOM, rules, defaults.marginBottom,
259                 messageCollector);
260 
261         return new SpriteLayoutProperties(alignment, marginLeft, marginRight, marginTop, marginBottom);
262     }
263 
264     /**
265      * Corrects sprite alignment if necessary based on the layout of the enclosing sprite image.
266      *
267      * @param spriteImageLayout
268      *            the sprite image layout
269      * @param alignment
270      *            the alignment
271      * @param messageCollector
272      *            the message collector
273      *
274      * @return the sprite alignment
275      */
276     private static SpriteAlignment correctAlignment(SpriteImageLayout spriteImageLayout, SpriteAlignment alignment,
277             MessageLog messageCollector) {
278         if (spriteImageLayout.equals(SpriteImageLayout.HORIZONTAL)) {
279             if (alignment.equals(SpriteAlignment.LEFT) || alignment.equals(SpriteAlignment.RIGHT)) {
280                 messageCollector.warning(MessageType.ONLY_TOP_OR_BOTTOM_ALIGNMENT_ALLOWED, alignment.value);
281                 return SpriteAlignment.TOP;
282             }
283         } else if (alignment.equals(SpriteAlignment.TOP) || alignment.equals(SpriteAlignment.BOTTOM)) {
284             messageCollector.warning(MessageType.ONLY_LEFT_OR_RIGHT_ALIGNMENT_ALLOWED, alignment.value);
285             return SpriteAlignment.LEFT;
286         }
287 
288         return alignment;
289     }
290 
291     /**
292      * Returns default alignment for given sprite image directive.
293      *
294      * @param spriteImageLayout
295      *            the sprite image layout
296      *
297      * @return the default alignment
298      */
299     private static SpriteAlignment getDefaultAlignment(SpriteImageLayout spriteImageLayout) {
300         if (spriteImageLayout.equals(SpriteImageLayout.HORIZONTAL)) {
301             return SpriteAlignment.TOP;
302         }
303         return SpriteAlignment.LEFT;
304     }

306     /**
307      * Parses margin value.
308      *
309      * @param marginRule
310      *            the margin rule
311      * @param rules
312      *            the rules
313      * @param defaultMargin
314      *            the default margin
315      * @param messageLog
316      *            the message log
317      *
318      * @return the margin
319      */
320     private static int getMargin(String marginRule, Map<String, CssProperty> rules, int defaultMargin,
321             MessageLog messageLog) {
322         if (!CssSyntaxUtils.hasNonBlankValue(rules, marginRule)) {
323             return defaultMargin;
324         }
325         final String rawMarginValue = rules.get(marginRule).value;
326         String marginValue = rawMarginValue;
327         if (marginValue.toLowerCase(Locale.ENGLISH).endsWith("px")) {
328             marginValue = marginValue.substring(0, marginValue.length() - 2);
329         }
330         try {
331             int marginIntValue = Integer.parseInt(marginValue);
332             if (marginIntValue < 0) {
333                 messageLog.warning(MessageType.IGNORING_NEGATIVE_MARGIN_VALUE, marginRule);
334                 marginIntValue = 0;
335             }
336 
337             return marginIntValue;
338         } catch (final NumberFormatException e) {
339             messageLog.warning(MessageType.CANNOT_PARSE_MARGIN_VALUE, rawMarginValue);
340             return 0;
341         }
342     }
