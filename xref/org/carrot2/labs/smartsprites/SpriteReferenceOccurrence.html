8   package org.carrot2.labs.smartsprites;
9   
10  import java.awt.image.BufferedImage;
11  
12  import org.carrot2.labs.smartsprites.SpriteImageDirective.SpriteImageLayout;
13  import org.carrot2.labs.smartsprites.SpriteLayoutProperties.SpriteAlignment;
14  import org.carrot2.util.BufferedImageUtils;
15  
16  /**
17   * Represents an occurrence of a {@link SpriteReferenceDirective} in a specific CSS file.
18   */
19  public class SpriteReferenceOccurrence extends SpriteDirectiveOccurrence {
20  
21      /** The directive. */
22      public final SpriteReferenceDirective spriteReferenceDirective;
23  
24      /** CSS file relative path to the individual image to be added to a sprite. */
25      public final String imagePath;
26  
27      /** Indicates whether the original css property has been marked as important. */
28      public final boolean important;
29  
30      /**
31       * Instantiates a new sprite reference occurrence.
32       *
33       * @param spriteReferenceDirective
34       *            the sprite reference directive
35       * @param imageFile
36       *            the image file
37       * @param cssFile
38       *            the css file
39       * @param line
40       *            the line
41       * @param important
42       *            the important
43       */
44      public SpriteReferenceOccurrence(SpriteReferenceDirective spriteReferenceDirective, String imageFile,
45              String cssFile, int line, boolean important) {
46          super(cssFile, line);
47          this.spriteReferenceDirective = spriteReferenceDirective;
48          this.imagePath = imageFile;
49          this.important = important;
50      }
51  
52      /**
53       * Computes the minimum width the individual image will need when rendering.
54       *
55       * @param image
56       *            the image
57       * @param layout
58       *            the layout
59       *
60       * @return the required width
61       */
62      public int getRequiredWidth(BufferedImage image, SpriteImageLayout layout) {
63          if (SpriteAlignment.REPEAT.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)
64                  && SpriteImageLayout.VERTICAL.equals(layout)) {
65              // Ignoring left/right margins on repeated
66              // images in vertically stacked sprites
67              return image.getWidth();
68          }
69          return image.getWidth() + spriteReferenceDirective.spriteLayoutProperties.marginLeft
70                  + spriteReferenceDirective.spriteLayoutProperties.marginRight;
71      }
72  
73      /**
74       * Computes the minimum height the individual image will need when rendering.
75       *
76       * @param image
77       *            the image
78       * @param layout
79       *            the layout
80       *
81       * @return the required height
82       */
83      public int getRequiredHeight(BufferedImage image, SpriteImageLayout layout) {
84          if (SpriteAlignment.REPEAT.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)
85                  && SpriteImageLayout.HORIZONTAL.equals(layout)) {
86              // Ignoring top/bottom margins on repeated
87              // images in horizontally lined sprites
88              return image.getHeight();
89          }
90          return image.getHeight() + spriteReferenceDirective.spriteLayoutProperties.marginTop
91                  + spriteReferenceDirective.spriteLayoutProperties.marginBottom;
92      }
93  
94      /**
95       * Renders the individual image, including margins and repeats if any.
96       *
97       * @param image
98       *            the individual image as read from the file
99       * @param layout
100      *            the layout the enclosing sprite
101      * @param dimension
102      *            height/width of a horizontal/vertical sprite
103      *
104      * @return the rendered individual image
105      */
106     public BufferedImage render(BufferedImage image, SpriteImageLayout layout, int dimension) {
107         final BufferedImage rendered;
108         if (SpriteImageLayout.VERTICAL.equals(layout)) {
109             rendered = new BufferedImage(dimension, getRequiredHeight(image, layout), BufferedImage.TYPE_4BYTE_ABGR);
110 
111             if (SpriteAlignment.LEFT.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
112                 BufferedImageUtils.drawImage(image, rendered,
113                         spriteReferenceDirective.spriteLayoutProperties.marginLeft,
114                         spriteReferenceDirective.spriteLayoutProperties.marginTop);
115             } else if (SpriteAlignment.RIGHT.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
116                 BufferedImageUtils.drawImage(image, rendered,
117                         dimension - spriteReferenceDirective.spriteLayoutProperties.marginRight - image.getWidth(),
118                         spriteReferenceDirective.spriteLayoutProperties.marginTop);
119             } else if (SpriteAlignment.CENTER.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
120                 BufferedImageUtils.drawImage(image, rendered, (rendered.getWidth() - image.getWidth()) / 2,
121                         spriteReferenceDirective.spriteLayoutProperties.marginTop);
122             } else {
123                 // Repeat, ignoring margin-left and margin-right
124                 for (int x = 0; x < dimension; x += image.getWidth()) {
125                     BufferedImageUtils.drawImage(image, rendered, x,
126                             spriteReferenceDirective.spriteLayoutProperties.marginTop);
127                 }
128             }
129         } else {
130             rendered = new BufferedImage(getRequiredWidth(image, layout), dimension, BufferedImage.TYPE_4BYTE_ABGR);
131 
132             if (SpriteAlignment.TOP.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
133                 BufferedImageUtils.drawImage(image, rendered,
134                         spriteReferenceDirective.spriteLayoutProperties.marginLeft,
135                         spriteReferenceDirective.spriteLayoutProperties.marginTop);
136             } else if (SpriteAlignment.BOTTOM.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
137                 BufferedImageUtils.drawImage(image, rendered,
138                         spriteReferenceDirective.spriteLayoutProperties.marginLeft,
139                         dimension - spriteReferenceDirective.spriteLayoutProperties.marginBottom - image.getHeight());
140             } else if (SpriteAlignment.CENTER.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
141                 BufferedImageUtils.drawImage(image, rendered,
142                         spriteReferenceDirective.spriteLayoutProperties.marginLeft,
143                         (rendered.getHeight() - image.getHeight()) / 2);
144             } else {
145                 // Repeat, ignoring margin-top and margin-bottom
146                 for (int y = 0; y < dimension; y += image.getHeight()) {
147                     BufferedImageUtils.drawImage(image, rendered,
148                             spriteReferenceDirective.spriteLayoutProperties.marginLeft, y);
149                 }
150             }
151         }
152         return rendered;
153     }
154 
155     /**
156      * Returns the {@link SpriteReferenceReplacement} corresponding to the occurrence, taking into account the layout
157      * the the enclosing sprite and the offset at which the individual image was rendered.
158      *
159      * @param layout
160      *            the layout
161      * @param offset
162      *            the offset
163      *
164      * @return the sprite reference replacement
165      */
166     public SpriteReferenceReplacement buildReplacement(SpriteImageLayout layout, int offset) {
167         if (SpriteImageLayout.VERTICAL.equals(layout)) {
168             String horizontalPosition;
169             if (SpriteAlignment.RIGHT.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
170                 horizontalPosition = "right";
171             } else if (SpriteAlignment.CENTER.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
172                 horizontalPosition = "center";
173             } else {
174                 horizontalPosition = "left";
175             }
176 
177             return new SpriteReferenceReplacement(this, offset, horizontalPosition);
178         }
179         String verticalPosition;
180         if (SpriteAlignment.BOTTOM.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
181             verticalPosition = "bottom";
182         } else if (SpriteAlignment.CENTER.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
183             verticalPosition = "center";
184         } else {
185             verticalPosition = "top";
186         }
187         return new SpriteReferenceReplacement(this, verticalPosition, offset);
188     }
189 }
