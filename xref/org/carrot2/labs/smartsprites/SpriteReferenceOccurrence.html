37  package org.carrot2.labs.smartsprites;
38  
39  import java.awt.image.BufferedImage;
40  
41  import org.carrot2.labs.smartsprites.SpriteImageDirective.SpriteImageLayout;
42  import org.carrot2.labs.smartsprites.SpriteLayoutProperties.SpriteAlignment;
43  import org.carrot2.util.BufferedImageUtils;
44  
45  /**
46   * Represents an occurrence of a {@link SpriteReferenceDirective} in a specific CSS file.
47   */
48  public class SpriteReferenceOccurrence extends SpriteDirectiveOccurrence {
49  
50      /** The directive. */
51      public final SpriteReferenceDirective spriteReferenceDirective;
52  
53      /** CSS file relative path to the individual image to be added to a sprite. */
54      public final String imagePath;
55  
56      /** Indicates whether the original css property has been marked as important. */
57      public final boolean important;
58  
59      /**
60       * Instantiates a new sprite reference occurrence.
61       *
62       * @param spriteReferenceDirective
63       *            the sprite reference directive
64       * @param imageFile
65       *            the image file
66       * @param cssFile
67       *            the css file
68       * @param line
69       *            the line
70       * @param important
71       *            the important
72       */
73      public SpriteReferenceOccurrence(SpriteReferenceDirective spriteReferenceDirective, String imageFile,
74              String cssFile, int line, boolean important) {
75          super(cssFile, line);
76          this.spriteReferenceDirective = spriteReferenceDirective;
77          this.imagePath = imageFile;
78          this.important = important;
79      }
80  
81      /**
82       * Computes the minimum width the individual image will need when rendering.
83       *
84       * @param image
85       *            the image
86       * @param layout
87       *            the layout
88       *
89       * @return the required width
90       */
91      public int getRequiredWidth(BufferedImage image, SpriteImageLayout layout) {
92          if (SpriteAlignment.REPEAT.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)
93                  && SpriteImageLayout.VERTICAL.equals(layout)) {
94              // Ignoring left/right margins on repeated
95              // images in vertically stacked sprites
96              return image.getWidth();
97          }
98          return image.getWidth() + spriteReferenceDirective.spriteLayoutProperties.marginLeft
99                  + spriteReferenceDirective.spriteLayoutProperties.marginRight;
100     }
101 
102     /**
103      * Computes the minimum height the individual image will need when rendering.
104      *
105      * @param image
106      *            the image
107      * @param layout
108      *            the layout
109      *
110      * @return the required height
111      */
112     public int getRequiredHeight(BufferedImage image, SpriteImageLayout layout) {
113         if (SpriteAlignment.REPEAT.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)
114                 && SpriteImageLayout.HORIZONTAL.equals(layout)) {
115             // Ignoring top/bottom margins on repeated
116             // images in horizontally lined sprites
117             return image.getHeight();
118         }
119         return image.getHeight() + spriteReferenceDirective.spriteLayoutProperties.marginTop
120                 + spriteReferenceDirective.spriteLayoutProperties.marginBottom;
121     }
122 
123     /**
124      * Renders the individual image, including margins and repeats if any.
125      *
126      * @param image
127      *            the individual image as read from the file
128      * @param layout
129      *            the layout the enclosing sprite
130      * @param dimension
131      *            height/width of a horizontal/vertical sprite
132      *
133      * @return the rendered individual image
134      */
135     public BufferedImage render(BufferedImage image, SpriteImageLayout layout, int dimension) {
136         final BufferedImage rendered;
137         if (SpriteImageLayout.VERTICAL.equals(layout)) {
138             rendered = new BufferedImage(dimension, getRequiredHeight(image, layout), BufferedImage.TYPE_4BYTE_ABGR);
139 
140             if (SpriteAlignment.LEFT.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
141                 BufferedImageUtils.drawImage(image, rendered,
142                         spriteReferenceDirective.spriteLayoutProperties.marginLeft,
143                         spriteReferenceDirective.spriteLayoutProperties.marginTop);
144             } else if (SpriteAlignment.RIGHT.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
145                 BufferedImageUtils.drawImage(image, rendered,
146                         dimension - spriteReferenceDirective.spriteLayoutProperties.marginRight - image.getWidth(),
147                         spriteReferenceDirective.spriteLayoutProperties.marginTop);
148             } else if (SpriteAlignment.CENTER.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
149                 BufferedImageUtils.drawImage(image, rendered, (rendered.getWidth() - image.getWidth()) / 2,
150                         spriteReferenceDirective.spriteLayoutProperties.marginTop);
151             } else {
152                 // Repeat, ignoring margin-left and margin-right
153                 for (int x = 0; x < dimension; x += image.getWidth()) {
154                     BufferedImageUtils.drawImage(image, rendered, x,
155                             spriteReferenceDirective.spriteLayoutProperties.marginTop);
156                 }
157             }
158         } else {
159             rendered = new BufferedImage(getRequiredWidth(image, layout), dimension, BufferedImage.TYPE_4BYTE_ABGR);
160 
161             if (SpriteAlignment.TOP.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
162                 BufferedImageUtils.drawImage(image, rendered,
163                         spriteReferenceDirective.spriteLayoutProperties.marginLeft,
164                         spriteReferenceDirective.spriteLayoutProperties.marginTop);
165             } else if (SpriteAlignment.BOTTOM.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
166                 BufferedImageUtils.drawImage(image, rendered,
167                         spriteReferenceDirective.spriteLayoutProperties.marginLeft,
168                         dimension - spriteReferenceDirective.spriteLayoutProperties.marginBottom - image.getHeight());
169             } else if (SpriteAlignment.CENTER.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
170                 BufferedImageUtils.drawImage(image, rendered,
171                         spriteReferenceDirective.spriteLayoutProperties.marginLeft,
172                         (rendered.getHeight() - image.getHeight()) / 2);
173             } else {
174                 // Repeat, ignoring margin-top and margin-bottom
175                 for (int y = 0; y < dimension; y += image.getHeight()) {
176                     BufferedImageUtils.drawImage(image, rendered,
177                             spriteReferenceDirective.spriteLayoutProperties.marginLeft, y);
178                 }
179             }
180         }
181         return rendered;
182     }
183 
184     /**
185      * Returns the {@link SpriteReferenceReplacement} corresponding to the occurrence, taking into account the layout
186      * the the enclosing sprite and the offset at which the individual image was rendered.
187      *
188      * @param layout
189      *            the layout
190      * @param offset
191      *            the offset
192      *
193      * @return the sprite reference replacement
194      */
195     public SpriteReferenceReplacement buildReplacement(SpriteImageLayout layout, int offset) {
196         if (SpriteImageLayout.VERTICAL.equals(layout)) {
197             String horizontalPosition;
198             if (SpriteAlignment.RIGHT.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
199                 horizontalPosition = "right";
200             } else if (SpriteAlignment.CENTER.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
201                 horizontalPosition = "center";
202             } else {
203                 horizontalPosition = "left";
204             }
205 
206             return new SpriteReferenceReplacement(this, offset, horizontalPosition);
207         }
208         String verticalPosition;
209         if (SpriteAlignment.BOTTOM.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
210             verticalPosition = "bottom";
211         } else if (SpriteAlignment.CENTER.equals(spriteReferenceDirective.spriteLayoutProperties.alignment)) {
212             verticalPosition = "center";
213         } else {
214             verticalPosition = "top";
215         }
216         return new SpriteReferenceReplacement(this, verticalPosition, offset);
217     }
