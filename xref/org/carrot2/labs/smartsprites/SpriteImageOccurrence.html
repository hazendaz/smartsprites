8   package org.carrot2.labs.smartsprites;
9   
10  /**
11   * Describes an occurrence of a {@link SpriteImageDirective} in a specific CSS file.
12   */
13  public class SpriteImageOccurrence extends SpriteDirectiveOccurrence {
14  
15      /** The sprite image directive. */
16      public final SpriteImageDirective spriteImageDirective;
17  
18      /**
19       * Instantiates a new sprite image occurrence.
20       *
21       * @param spriteImageDirective
22       *            the sprite image directive
23       * @param cssFile
24       *            the css file
25       * @param line
26       *            the line
27       */
28      public SpriteImageOccurrence(SpriteImageDirective spriteImageDirective, String cssFile, int line) {
29          super(cssFile, line);
30          this.spriteImageDirective = spriteImageDirective;
31      }
32  }
