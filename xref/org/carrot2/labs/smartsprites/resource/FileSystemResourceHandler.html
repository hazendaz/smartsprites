37  package org.carrot2.labs.smartsprites.resource;
38  
39  import java.io.File;
40  import java.io.FileInputStream;
41  import java.io.FileOutputStream;
42  import java.io.IOException;
43  import java.io.InputStream;
44  import java.io.InputStreamReader;
45  import java.io.OutputStream;
46  import java.io.OutputStreamWriter;
47  import java.io.Reader;
48  import java.io.UnsupportedEncodingException;
49  import java.io.Writer;
50  import java.nio.charset.Charset;
51  
52  import org.apache.commons.io.FilenameUtils;
53  import org.carrot2.labs.smartsprites.SmartSpritesParameters;
54  import org.carrot2.labs.smartsprites.message.Message;
55  import org.carrot2.labs.smartsprites.message.Message.MessageType;
56  import org.carrot2.labs.smartsprites.message.MessageLog;
57  import org.carrot2.util.FileUtils;
58  import org.carrot2.util.StringUtils;
59  
60  /**
61   * This class defines the resource handler which manage resources from the file system.
62   *
63   * @author Ibrahim Chaehoi
64   * @author Stanislaw Osinski
65   */
66  public class FileSystemResourceHandler implements ResourceHandler {
67  
68      /** The message log. */
69      private final MessageLog messageLog;
70  
71      /** The root directory. */
72      private final String documentRootDir;
73  
74      /** The charset to assume in the {@link #getResourceAsReader(String)} method. */
75      private final Charset charset;
76  
77      /**
78       * Creates a new {@link FileSystemResourceHandler}.
79       *
80       * @param documentRootDirPath
81       *            the document root directory path, can be <code>null</code>
82       * @param charset
83       *            the charset to assume in the {@link #getResourceAsReader(String)} method
84       * @param messageLog
85       *            the message log
86       */
87      public FileSystemResourceHandler(String documentRootDirPath, String charset, MessageLog messageLog) {
88          this.documentRootDir = documentRootDirPath;
89          this.messageLog = messageLog;
90          this.charset = Charset.forName(charset);
91      }
92  
93      @Override
94      public InputStream getResourceAsInputStream(String path) throws IOException {
95          return new FileInputStream(FileUtils.getCanonicalOrAbsoluteFile(path));
96      }
97  
98      @Override
99      public Reader getResourceAsReader(String path) throws IOException {
100         try {
101             return new InputStreamReader(getResourceAsInputStream(path), charset);
102         } catch (UnsupportedEncodingException e) {
103             // Should not happen as we're checking the charset in constructor
104             throw new RuntimeException(e);
105         }
106     }
107 
108     @Override
109     public OutputStream getResourceAsOutputStream(String path) throws IOException {
110         // Create directories if needed
111         final File parentFile = new File(path).getParentFile();
112         if (!parentFile.exists() && !parentFile.mkdirs()) {
113             messageLog.warning(Message.MessageType.CANNOT_CREATE_DIRECTORIES, parentFile.getPath());
114         }
115         return new FileOutputStream(FileUtils.getCanonicalOrAbsoluteFile(path));
116     }
117 
118     @Override
119     public Writer getResourceAsWriter(String path) throws IOException {
120         try {
121             return new OutputStreamWriter(getResourceAsOutputStream(path), charset);
122         } catch (UnsupportedEncodingException e) {
123             // Should not happen as we're checking the charset in constructor
124             throw new RuntimeException(e);
125         }
126     }
127 
128     /**
129      * This implementation detects if the resource path starts with a "/" and resolves such resources against the
130      * provided {@link SmartSpritesParameters#getDocumentRootDir()} directory.
131      */
132     @Override
133     public String getResourcePath(String baseFile, String filePath) {
134         if (!filePath.startsWith("/")) {
135             return FilenameUtils.concat(FilenameUtils.getFullPath(baseFile), filePath);
136         }
137         if (StringUtils.isNotBlank(documentRootDir)) {
138             return FilenameUtils.concat(documentRootDir, filePath.substring(1));
139         }
140         messageLog.warning(MessageType.ABSOLUTE_PATH_AND_NO_DOCUMENT_ROOT, filePath);
141         return "";
142     }
143 }
