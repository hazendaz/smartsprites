8   package org.carrot2.labs.smartsprites;
9   
10  import static org.carrot2.labs.test.Assertions.assertThat;
11  import static org.junit.jupiter.api.Assertions.assertEquals;
12  
13  import org.carrot2.labs.smartsprites.css.CssSyntaxUtils;
14  import org.carrot2.labs.smartsprites.message.Message;
15  import org.carrot2.labs.smartsprites.resource.FileSystemResourceHandler;
16  import org.junit.jupiter.api.BeforeEach;
17  import org.junit.jupiter.api.Test;
18  
19  /**
20   * Test cases for {@link SpriteDirectiveOccurrenceCollector}.
21   */
22  class SpriteDirectiveOccurrenceCollectorTest extends TestWithMemoryMessageSink {
23  
24      /** The sprite directive occurrence collector. */
25      SpriteDirectiveOccurrenceCollector spriteDirectiveOccurrenceCollector;
26  
27      /**
28       * Prepare.
29       */
30      @BeforeEach
31      void prepare() {
32          spriteDirectiveOccurrenceCollector = new SpriteDirectiveOccurrenceCollector(messageLog,
33                  new FileSystemResourceHandler(null, SmartSpritesParameters.DEFAULT_CSS_FILE_ENCODING, messageLog));
34      }
35  
36      /**
37       * Test sprite image directive extraction one directive complex.
38       */
39      @Test
40      void testSpriteImageDirectiveExtractionOneDirectiveComplex() {
41          final String spriteDirective = "sprite: sprite; sprite-image-url: url('../sprite.png'); sprite-image-layout: vertical";
42          final String css = ".test { margin-top: 10px }\n/* some comment */\n" + "/* " + spriteDirective + " */";
43  
44          assertEquals(spriteDirective, SpriteDirectiveOccurrenceCollector.extractSpriteImageDirectiveString(css));
45      }
46  
47      /**
48       * Test sprite image directive extraction one directive simple.
49       */
50      @Test
51      void testSpriteImageDirectiveExtractionOneDirectiveSimple() {
52          final String spriteDirective = "sprite: sprite";
53          final String css = "/* " + spriteDirective + " */";
54  
55          assertEquals(spriteDirective, SpriteDirectiveOccurrenceCollector.extractSpriteImageDirectiveString(css));
56      }
57  
58      /**
59       * Test sprite image directive extraction more directives.
60       */
61      @Test
62      void testSpriteImageDirectiveExtractionMoreDirectives() {
63          final String spriteDirective1 = "sprite: sprite; sprite-image-url: url('../sprite.png'); sprite-image-layout: vertical";
64          final String spriteDirective2 = "sprite: sprite2; sprite-image-url: url('../sprite2.png'); sprite-image-layout: horizontal";
65          final String css = ".test { margin-top: 10px }\n/* some comment */\n" + "/* " + spriteDirective1 + " */\n"
66                  + ".rule { float: left }\n" + "/*** \t" + spriteDirective2 + " \t **/";
67  
68          assertEquals(spriteDirective1, SpriteDirectiveOccurrenceCollector.extractSpriteImageDirectiveString(css));
69      }
70  
71      /**
72       * Test sprite reference directive extraction.
73       */
74      @Test
75      void testSpriteReferenceDirectiveExtraction() {
76          final String spriteDirective = "sprite-ref: sprite; sprite-alignment: repeat";
77          final String css = "background-image: url('../img/img.png'); /** " + spriteDirective + " */";
78  
79          assertEquals(spriteDirective, SpriteDirectiveOccurrenceCollector.extractSpriteReferenceDirectiveString(css));
80      }
81  
82      /**
83       * Test sprite reference image url extraction.
84       */
85      @Test
86      void testSpriteReferenceImageUrlExtraction() {
87          final String spriteDirective = "sprite-ref: sprite; sprite-alignment: repeat";
88          final String css = "background-image: url('../img/img.png'); /** " + spriteDirective + " */";
89  
90          assertEquals("../img/img.png", CssSyntaxUtils
91                  .unpackUrl(spriteDirectiveOccurrenceCollector.extractSpriteReferenceCssProperty(css).value, null));
92      }
93  
94      /**
95       * Test sprite reference image url extraction no background image.
96       */
97      @Test
98      void testSpriteReferenceImageUrlExtractionNoBackgroundImage() {
99          final String spriteDirective = "sprite-ref: sprite; sprite-alignment: repeat";
100         final String css = "background-imagez: url('../img/img.png'); /** " + spriteDirective + " */";
101 
102         assertEquals(null, spriteDirectiveOccurrenceCollector.extractSpriteReferenceCssProperty(css));
103 
104         assertThat(messages).isEquivalentTo(new Message(Message.MessageLevel.WARN,
105                 Message.MessageType.NO_BACKGROUND_IMAGE_RULE_NEXT_TO_SPRITE_REFERENCE_DIRECTIVE, null, 0, css));
106     }
107 
108     /**
109      * Test sprite reference image url extraction more rules.
110      */
111     @Test
112     void testSpriteReferenceImageUrlExtractionMoreRules() {
113         final String spriteDirective = "sprite-ref: sprite; sprite-alignment: repeat";
114         final String css = "color: red; background-image: url('../img/img.png'); /** " + spriteDirective + " */";
115 
116         assertEquals(null, spriteDirectiveOccurrenceCollector.extractSpriteReferenceCssProperty(css));
117 
118         assertThat(messages).isEquivalentTo(new Message(Message.MessageLevel.WARN,
119                 Message.MessageType.MORE_THAN_ONE_RULE_NEXT_TO_SPRITE_REFERENCE_DIRECTIVE, null, 0, css));
120     }
121 }
