8   package org.carrot2.labs.smartsprites.message;
9   
10  import static org.junit.jupiter.api.Assertions.assertEquals;
11  import static org.junit.jupiter.api.Assertions.assertTrue;
12  
13  import org.carrot2.labs.smartsprites.message.Message.MessageLevel;
14  import org.carrot2.labs.smartsprites.message.Message.MessageType;
15  import org.junit.jupiter.api.Test;
16  
17  /**
18   * Test cases for {@link Message}.
19   */
20  class MessageTest {
21  
22      /**
23       * Warn creates message with warn level and no css path.
24       */
25      @Test
26      void warnCreatesMessageWithWarnLevel() {
27          Message msg = Message.warn(MessageType.GENERIC, "test warning");
28          assertEquals(MessageLevel.WARN, msg.level);
29          assertEquals(MessageType.GENERIC, msg.type);
30          assertEquals(1, msg.arguments.length);
31          assertEquals("test warning", msg.arguments[0]);
32      }
33  
34      /**
35       * Error creates message with error level.
36       */
37      @Test
38      void errorCreatesMessageWithErrorLevel() {
39          Message msg = Message.error(MessageType.GENERIC, "test error");
40          assertEquals(MessageLevel.ERROR, msg.level);
41          assertEquals(MessageType.GENERIC, msg.type);
42          assertEquals(1, msg.arguments.length);
43          assertEquals("test error", msg.arguments[0]);
44      }
45  
46      /**
47       * To string with no css path contains level and message.
48       */
49      @Test
50      void toStringWithNoCssPath() {
51          Message msg = Message.warn(MessageType.GENERIC, "hello");
52          String result = msg.toString();
53          assertTrue(result.startsWith("WARN: "), "Should start with level: " + result);
54          assertTrue(result.contains("hello"), "Should contain formatted message: " + result);
55      }
56  
57      /**
58       * To string with css path includes file and line info.
59       */
60      @Test
61      void toStringWithCssPathIncludesFileInfo() {
62          Message msg = new Message(MessageLevel.ERROR, MessageType.GENERIC, "test.css", 5, "msg");
63          String result = msg.toString();
64          assertTrue(result.contains("test.css"), "Should contain css path: " + result);
65          // line is stored 0-based, displayed 1-based
66          assertTrue(result.contains("line: 6"), "Should contain line number (1-based): " + result);
67      }
68  
69      /**
70       * Get formatted message formats arguments.
71       */
72      @Test
73      void getFormattedMessageWithArguments() {
74          Message msg = Message.warn(MessageType.CANNOT_NOT_LOAD_IMAGE, "file.png", "IOException");
75          String formatted = msg.getFormattedMessage();
76          assertTrue(formatted.contains("file.png"), "Should contain file name: " + formatted);
77          assertTrue(formatted.contains("IOException"), "Should contain exception: " + formatted);
78      }
79  
80      /**
81       * Message level comparator orders levels correctly.
82       */
83      @Test
84      void messageLevelComparatorOrdersLevelsCorrectly() {
85          assertTrue(MessageLevel.COMPARATOR.compare(MessageLevel.INFO, MessageLevel.WARN) < 0);
86          assertTrue(MessageLevel.COMPARATOR.compare(MessageLevel.WARN, MessageLevel.INFO) > 0);
87          assertEquals(0, MessageLevel.COMPARATOR.compare(MessageLevel.INFO, MessageLevel.INFO));
88      }
89  
90      /**
91       * Message level comparator orders error above warn.
92       */
93      @Test
94      void messageLevelComparatorOrdersErrorAboveWarn() {
95          assertTrue(MessageLevel.COMPARATOR.compare(MessageLevel.WARN, MessageLevel.ERROR) < 0);
96          assertTrue(MessageLevel.COMPARATOR.compare(MessageLevel.ERROR, MessageLevel.STATUS) < 0);
97      }
98  
99      /**
100      * Message constructor copies arguments array defensively.
101      */
102     @Test
103     void messageConstructorCopiesArguments() {
104         Object[] args = { "arg1", "arg2" };
105         Message msg = new Message(MessageLevel.INFO, MessageType.GENERIC, null, 0, args);
106         // Modify original array - should not affect message
107         args[0] = "changed";
108         assertEquals("arg1", msg.arguments[0], "Arguments should be copied defensively");
109     }
110 
111     /**
112      * To string with info level and no css path.
113      */
114     @Test
115     void toStringInfoLevelNoCssPath() {
116         Message msg = new Message(MessageLevel.INFO, MessageType.PROCESSING_COMPLETED, null, 0, 100L);
117         String result = msg.toString();
118         assertTrue(result.startsWith("INFO: "), "Should start with INFO: " + result);
119     }
120 
121     /**
122      * Warn with no arguments produces valid message.
123      */
124     @Test
125     void warnWithNoArguments() {
126         Message msg = Message.warn(MessageType.SPRITE_ID_NOT_FOUND);
127         assertEquals(MessageLevel.WARN, msg.level);
128         assertEquals(0, msg.arguments.length);
129     }
130 
131     /**
132      * Error with no arguments produces valid message.
133      */
134     @Test
135     void errorWithNoArguments() {
136         Message msg = Message.error(MessageType.EITHER_ROOT_DIR_OR_CSS_FILES_IS_REQUIRED);
137         assertEquals(MessageLevel.ERROR, msg.level);
138         assertEquals(0, msg.arguments.length);
139     }
140 }
