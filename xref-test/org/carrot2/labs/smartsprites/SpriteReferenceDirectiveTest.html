193     /**
194      * Check mismatched alignment.
195      *
196      * @param sprite
197      *            the sprite
198      * @param alignment
199      *            the alignment
200      * @param correctedAlignment
201      *            the corrected alignment
202      * @param message
203      *            the message
204      */
205     private void checkMismatchedAlignment(String sprite, final String alignment, SpriteAlignment correctedAlignment,
206             MessageType message) {
207         final SpriteReferenceDirective directive = SpriteReferenceDirective.parse(
208                 "sprite-ref: " + sprite + "; sprite-alignment: " + alignment, SPRITE_IMAGE_DIRECTIVES, messageLog);
209 
210         assertNotNull(directive);
211         assertEquals(sprite, directive.spriteRef);
212         assertEquals(correctedAlignment, directive.spriteLayoutProperties.alignment);
213         assertEquals(0, directive.spriteLayoutProperties.marginLeft);
214         assertEquals(0, directive.spriteLayoutProperties.marginRight);
215         assertEquals(0, directive.spriteLayoutProperties.marginTop);
216         assertEquals(0, directive.spriteLayoutProperties.marginBottom);
217 
218         assertThat(messages).isEquivalentTo(new Message(Message.MessageLevel.WARN, message, null, 0, alignment));
219     }
